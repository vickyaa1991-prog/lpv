#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <omp.h>

#define N 10000

int a[N], b[N], tmp[N];

void fill(int* arr) {
    srand(42);
    for (int i = 0; i < N; i++) arr[i] = rand() % 10000;
}

void copy(int* src, int* dst) {
    for (int i = 0; i < N; i++) dst[i] = src[i];
}

/* ---------- Bubble Sort ---------- */
void seqBubble(int* arr, int n) {
    for (int i = 0; i < n-1; i++)
        for (int j = 0; j < n-i-1; j++)
            if (arr[j] > arr[j+1]) { int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t; }
}

void parBubble(int* arr, int n) {
    for (int i = 0; i < n; i++) {
        int phase = i % 2;
        #pragma omp parallel for shared(arr)
        for (int j = phase; j < n-1; j += 2)
            if (arr[j] > arr[j+1]) { int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t; }
    }
}

/* ---------- Merge Sort ---------- */
void merge(int* arr, int l, int m, int r) {
    int n1 = m-l+1, n2 = r-m;
    int L[n1], R[n2];
    for (int i = 0; i < n1; i++) L[i] = arr[l+i];
    for (int i = 0; i < n2; i++) R[i] = arr[m+1+i];
    int i = 0, j = 0, k = l;
    while (i < n1 && j < n2) arr[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void seqMerge(int* arr, int l, int r) {
    if (l < r) {
        int m = (l+r)/2;
        seqMerge(arr, l, m);
        seqMerge(arr, m+1, r);
        merge(arr, l, m, r);
    }
}

void parMerge(int* arr, int l, int r, int depth) {
    if (l < r) {
        int m = (l+r)/2;
        if (depth > 0) {
            #pragma omp parallel sections
            {
                #pragma omp section
                parMerge(arr, l, m, depth-1);
                #pragma omp section
                parMerge(arr, m+1, r, depth-1);
            }
        } else {
            seqMerge(arr, l, m);
            seqMerge(arr, m+1, r);
        }
        merge(arr, l, m, r);
    }
}

/* ---------- Main ---------- */
int main() {
    fill(a);
    double t1, t2;

    printf("Array size: %d\n\n", N);

    copy(a, b);
    t1 = omp_get_wtime();
    seqBubble(b, N);
    t2 = omp_get_wtime();
    printf("Sequential Bubble Sort : %.4f sec\n", t2-t1);

    copy(a, b);
    t1 = omp_get_wtime();
    parBubble(b, N);
    t2 = omp_get_wtime();
    printf("Parallel   Bubble Sort : %.4f sec\n", t2-t1);

    copy(a, b);
    t1 = omp_get_wtime();
    seqMerge(b, 0, N-1);
    t2 = omp_get_wtime();
    printf("\nSequential Merge  Sort : %.4f sec\n", t2-t1);

    copy(a, b);
    t1 = omp_get_wtime();
    #pragma omp parallel
    #pragma omp single
    parMerge(b, 0, N-1, 3);
    t2 = omp_get_wtime();
    printf("Parallel   Merge  Sort : %.4f sec\n", t2-t1);

    return 0;
}
