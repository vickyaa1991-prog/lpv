#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define N 1000000

int arr[N];

void fill() {
    srand(42);
    for (int i = 0; i < N; i++) arr[i] = rand() % 10000;
}

/* ---------- Sequential ---------- */
void seqOps(int* a, int n) {
    int mn = a[0], mx = a[0];
    long long sum = 0;

    for (int i = 0; i < n; i++) {
        if (a[i] < mn) mn = a[i];
        if (a[i] > mx) mx = a[i];
        sum += a[i];
    }

    printf("Sequential:\n");
    printf("  Min     = %d\n", mn);
    printf("  Max     = %d\n", mx);
    printf("  Sum     = %lld\n", sum);
    printf("  Average = %.2f\n", (double)sum / n);
}

/* ---------- Parallel Reduction ---------- */
void parOps(int* a, int n) {
    int mn = a[0], mx = a[0];
    long long sum = 0;

    #pragma omp parallel for reduction(min:mn) reduction(max:mx) reduction(+:sum)
    for (int i = 0; i < n; i++) {
        if (a[i] < mn) mn = a[i];
        if (a[i] > mx) mx = a[i];
        sum += a[i];
    }

    printf("Parallel:\n");
    printf("  Min     = %d\n", mn);
    printf("  Max     = %d\n", mx);
    printf("  Sum     = %lld\n", sum);
    printf("  Average = %.2f\n", (double)sum / n);
}

/* ---------- Main ---------- */
int main() {
    fill();
    double t1, t2;

    printf("Array size: %d\n\n", N);

    t1 = omp_get_wtime();
    seqOps(arr, N);
    t2 = omp_get_wtime();
    printf("  Time    = %.6f sec\n\n", t2 - t1);

    t1 = omp_get_wtime();
    parOps(arr, N);
    t2 = omp_get_wtime();
    printf("  Time    = %.6f sec\n", t2 - t1);

    return 0;
}
